概率计算

有goldtree和greentree，获得它们的概率分别为5%和95%

这是JS代码:
									function Percent(oArr) {
							            this.sum = 0;
							            this.rand = 0;
							            this.result = 0;
							            this.oArr = oArr
							            this.count()
							        }
							        Percent.prototype = {
							            count: function() {
							                for(var i in this.oArr) {
							                    this.sum += this.oArr[i][0];
							                }
							                for(var i in this.oArr) {
							                    this.rand = Math.floor(Math.random() * this.sum + 1);
							                    if(this.oArr[i][0] >= this.rand) {
							                        this.result = this.oArr[i][1];
							                        break;
							                    } else {
							                        this.sum -= this.oArr[i][0]
							                    }
							                }
							                return this.result;
							            }
							        }
							        var aa = new Percent({
							            "1": [5, "goldTree"],
							            "2": [95, "greenTree"]
							        })
							        console.log(aa.result)
